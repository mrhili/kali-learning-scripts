# Simple ip generation and verry flexible


Example

./ipgen.sh 192.168.1.1-6 output.txt

output:
    192.168.1.1
    192.168.1.2
    192.168.1.3
    192.168.1.4
    192.168.1.5
    192.168.1.6
    
# Dont forget to chmod u+x the script before using
